// 여기 설명부터 월요일에 다시
// fname부터

void save(const std::string& fname)
{
	std::ifstream in{ fname };
	std::ofstream out{ "2026년 1학기 STL 월56화78.txt", std::ios::app};
	std::copy(std::istreambuf_iterator<char>{in}, {},
		std::ostream_iterator<char>{out});
}
