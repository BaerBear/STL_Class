// STL 컨테이너 - Containers are objects that store other object
// Sequence containers - vector, list, deque
// - array<T,N> 유일하게 컴파일 타임에 size 결정 - STACK, DATA
// - vector<T> - cache hit율이 높아 고속 데이터 처리에 유리
// - list<T> - 아무데서나 원소 추가/삭제  O(1)
// - deque<T> - vector와 list의 장점을 갖는 컨테이너이다.
//==========================================================================================
#include "save.h"
#include "ZString.h"

extern bool 관찰; //관찰하려면 true

/*
벡터는 연속으로 ZString을 쓴다고 하면 24바이트가 연속이 되는데
리스트라면 앞 뒤로 next prev로 8바이트가 붙여서 메모리가 커지게 된다
덱은 벡터처럼 연속으로 되지만 앞뒤로 next prev가 붙는 것을 볼 수 있다 capacity는 없다 대신 다른 것들이 많아서 엄청크다 32에서 40정도
덱은 어떤 점이 좋은지 생각을 해봐야한ㄷ
stl 컨테이너는 MSVC인데 deque이 이용하는 메모리 크기를 전체 16바이트로 잡고 있음 덱을 이용한다고해서 리스트를 쓸때와 다를게 없음
ZString을 쓴다면 24니까 연속이 안되는데
g++ clang에서는 더 크게 잡고있다 512나 1024나 그래서 장점이 나온다

가장 많이 담는 컨테이너는 전부 다 push back한다면(원소를 한 개씩 추가한다면)
vector -> 무조건 꼴지님 가용한 메모리가 있다면 하나씩 채워나가면 50%채우는 순간 나머지는 못채우고 터져버린다 가장 꼴지임
deque -> list보다 deque보다 더 많이 담아야하는 데 msvc에서는 리스트와 별 차이가 없다 젤 많이 담기 가능
list ->
stl deque의 메모리 블럭 크기는 어떻게 되냐
c++은 512바이트
clang는 4096바이트
msvc는 16바이틀르 담는다 이게 비주얼 스튜디오
*/

class Test {
	char c[1024 * 1024];
};

	{
		std::vector<Test> v;

		size_t cnt{};
		while (true) {
			try {
				v.emplace_back();
			}
			catch (...) {//... 이런 3개의 점은 타원들(ellipses)라고 부른다
				std::cout << "벡터 - " << v.size() << "개 담음" << std::endl;
				break;
			}
			if (not(++cnt % 1000))
				std::cout << ".";

		}


		//벡터는 약 78억개인데 100%를 다 쓴다면 156억개까지 가능하다는 것이다 벡터는 가용된 메모리를 50%밖에 못쓰기 때문에
		//저런식으로 하면 벡터는 27318개를 담는다
	}

	{
		std::deque<Test> v;

		size_t cnt{};
		while (true) {
			try {
				v.emplace_back();
			}
			catch (...) {//... 이런 3개의 점은 타원들(ellipses)라고 부른다
				std::cout << "덱 - " << v.size() << "개 담음" << std::endl;
				break;
			}
			if (not(++cnt % 1000))
				std::cout << ".";

		}

		//저런식으로 하면 덱는 53625개를 담는다 벡터의 두배이다
		
	}
	{
		//std::list<int> v;//이런식으로 써서 돌리면 컴퓨터가 터진다 그 이유는 벡터랑 다르게 끝 까지먹는 것이기에 컴퓨터가 도저히 못버틴다
		//한다면 윈도우가 복구가 안되고 터진다.
