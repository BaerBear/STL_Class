	ZString(const char* s) {
		len = strlen(s);						// 글자 수 세서 len에 저장
		p = std::make_unique<char[]>(len);		// 글자 수 만큼 unique_ptr로 char 배열을 관리

			os << *(zs.p.get() + i);
		}
		return os;
	}

private:
	size_t len{};
	std::unique_ptr<char[]> p{};

};
