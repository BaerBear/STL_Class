#include <iostream>

class Dog;			// 전방선언 forward declaration. Dog라는 이름의 클래스가 나중에 정의될 것이다. 
					// change 함수에서 Dog&를 사용하기 위해 필요하다.

// 함수명이 같아도 인자가 다르면 컴파일러가 알아서 조작한다.
// 이걸 name mangling이라고 하는데, 컴파일러가 함수 이름과 인자 타입을 조합해서 고유한 이름을 만들어내는 것이다.
void change(Dog&, Dog&);
void change(int&, int&);

class Dog {
public:
	Dog() = default;
	explicit Dog(int n) : num { n } {};
	// 이런 경우는 관리할 자원이 (ex. 파일, 메모리, 소켓 등) 없어서 굳이 소멸자를 만들 이유가 없다 
	
	// 이 코드는 사용하지 말자
	//operator int () { // Dog를 int로 변환시킬 수 있음. 일반적인 방법은 아니고 소개만
	//	return num;
	//}

private:
	int num;

	friend std::ostream& operator<<(std::ostream& os, const Dog& dog) {
		return os << dog.num;
	}
};

//--------
int main() 
//--------
{
	// 컴파일 할 때 모든 주소가 결정되는데 유일하게 결정할 수 없는게 하나 있다.
	// new int; new로 동적 할당한 메모리는 런타임에 주소가 결정된다. 그래서 new로 동적 할당한 메모리는 컴파일 타임에 주소를 알 수 없다.
	int* p = new int; // 런타임에 주소가 결정되기 때문에 이 주소를 지정해주는 포인터 p는 런타임에 주소가 결정되는 포인터이다.

	// [문제] 의도대로 실행되게 하자. (시험문제)
	{
		// using Dog = int;		// #define Dog int 이건 사용하면 안된다
		// Dog는 class로 작성하자.
		// Dog a = 1;		// Dog a = (int)1 / int가 Dog로 변환하기 위한 과정을 컴파일러가 고민함. 
							// Dog a = { (int)1 } 와 같이 생성자로 넘기는걸 자동으로 해줌.
		Dog a{ 1 }, b{ 2 };
		change(a, b);
		std::cout << a << ", " << b << std::endl;		// 2, 1이라고 출력되어야 함
	}

	{
		int a{ 1 }, b{ 2 };
		change(a, b);
		std::cout << a << ", " << b << std::endl;		// 2, 1이라고 출력되어야 함
		//std::cout.operator << (a) << std::endl;
	}

	save("메인.cpp"); 
}

void change(Dog& p, Dog& q) {
	// std::swap(p, q);
	Dog temp{ p };
	p.operator = (q);	// == p = q;
	q.operator = (temp); // == q = temp;
}

void change(int& p, int& q) {
	std::swap(p, q);
}
